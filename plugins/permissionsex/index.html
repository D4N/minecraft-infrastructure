Example permissions.yml¶

schema-version: 1
groups:
  default:
    options:
      default: true
    permissions:
      - multiverse.portal.access.*
      - modifyworld.chat

  gamers_admins:
    inheritance:
      - commons_server_admin

  gamers_creative_builder:
    worlds:
      world:
        inheritance:
          - commons_world_builder
        permissions:
          - worldedit.*

  gamers_survival_builder:
    inheritance:
      - default
    worlds:
      world_survival:
        inheritance:
          - commons_world_builder


# commons Groups Collection of usecase specific Permissions

  commons_server_admin:
    inheritance:
      - commons_world_admins
    permissions:
      - '*'

  commons_world_builder:
    permissions:
      - modifyworld.*

  commons_world_admins:
    inheritance:
      - commons_worldeditor_builder
    permissions:
      - worldedit.*
      - worldguard.*
