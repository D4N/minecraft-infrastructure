Configuration¶

Die Grundlegende VM Config findet über CloudInit statt, alles nachfolgede wird durch Ansible gesteuert.

#cloud-config

package_upgrade: true

packages:
  - git
  - firewalld

users:
  - name: ansible
    gecos: Ansible Controll User
    sudo: ALL=(ALL) NOPASSWD:ALL
    ssh_import_id: ansiblemanagement
